import numpy as np

import pandas as pd
import matplotlib.pyplot as plt
from scipy import stats

    # Use a breakpoint in the code line below to debug your script.

if __name__ == '__main__':
    print_hi('PyCharm')


    # 读取数据集
    data = pd.read_csv('D:\iris.csv')
    attributes = []

    # 'sepal_length', 'sepal_width', 'petal_length', 'petal_width'
    # 将属性读到一个数组中
    for column in data.columns:
        if column!="Species_name":
            attributes.append(column)
    print(attributes)

    t=[]
    for attribute in attributes:
        mean = np.mean(data[attribute])
        median = np.median(data[attribute])
        mode = stats.mode(data[attribute])
        range = np.ptp(data[attribute])
        quartiles = np.quantile(data[attribute], [0.25, 0.5, 0.75])
        # iqr = np.iqr(data[attribute])

        print(attribute.capitalize() + ': Mean =', mean, ', Median =', median, ', Mode =', mode.mode[0], ', Range =', range,
            ', Quartiles =', quartiles, )
        t.append(data[attribute])
        # plt.boxplot(data[attribute])
    plt.boxplot(t)
    x=range(len(attributes))
    plt.xticks(x,attributes,rotation=45)

    plt.rcParams['font.sans-serif']=['SimHei']
    plt.title('统计')
    plt.show()
